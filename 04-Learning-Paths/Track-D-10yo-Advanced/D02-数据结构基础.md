## 🧠 核心原理简述（给家长）

**数据结构** 是组织和存储数据的方式。选对数据结构，程序效率能提升几十倍甚至上百倍。

四种最基础的数据结构：

- **列表 (List)**：像一排储物柜，按顺序存放东西，随时可以插入、删除、查找
- **字典 (Dictionary)**：像通讯录，用名字（键）查找电话号码（值），速度极快
- **栈 (Stack)**：像叠盘子，只能从最上面放和取，后放的先拿（后进先出）
- **队列 (Queue)**：像排队买票，先来的先走（先进先出）

为什么学数据结构：
- **竞赛核心**：NOI、AI创新挑战赛都考数据结构
- **算法基础**：排序、搜索、图论都离不开数据结构
- **效率提升**：同样的问题，选对结构能快 100 倍
- **思维训练**：培养逻辑思维和抽象能力

**玩法描述**:
1. 创建一个"书架管理系统"：
```python
class Bookshelf:
    def __init__(self):
        self.books = []
    
    def add_book(self, title):
        self.books.append(title)
        print(f"添加《{title}》，书架现有 {len(self.books)} 本书")
    
    def remove_book(self, title):
        if title in self.books:
            self.books.remove(title)
            print(f"取走《{title}》，书架剩余 {len(self.books)} 本书")
        else:
            print(f"书架上没有《{title}》")
    
    def find_book(self, title):
        if title in self.books:
            position = self.books.index(title) + 1
            print(f"《{title}》在第 {position} 个位置")
        else:
            print(f"找不到《{title}》")
    
    def show_all(self):
        print("书架上的书：")
        for i, book in enumerate(self.books, 1):
            print(f"  {i}. 《{book}》")

# 使用
shelf = Bookshelf()
shelf.add_book("Python入门")
shelf.add_book("算法图解")
shelf.add_book("数据结构")
shelf.show_all()
shelf.find_book("算法图解")
shelf.remove_book("Python入门")
shelf.show_all()
```

**玩法描述**:
1. 创建一个"同学通讯录"：
```python
class ContactBook:
    def __init__(self):
        self.contacts = {}
    
    def add_contact(self, name, phone, hobby):
        self.contacts[name] = {
            "phone": phone,
            "hobby": hobby
        }
        print(f"已添加 {name} 的信息")
    
    def find_contact(self, name):
        if name in self.contacts:
            info = self.contacts[name]
            print(f"{name}：")
            print(f"  电话：{info['phone']}")
            print(f"  爱好：{info['hobby']}")
        else:
            print(f"通讯录里没有 {name}")
    
    def list_all(self):
        print(f"共有 {len(self.contacts)} 个联系人：")
        for name, info in self.contacts.items():
            print(f"  {name} - {info['phone']}")
    
    def find_by_hobby(self, hobby):
        print(f"喜欢 {hobby} 的同学：")
        found = False
        for name, info in self.contacts.items():
            if info["hobby"] == hobby:
                print(f"  {name}")
                found = True
        if not found:
            print("  没找到")

# 使用
contacts = ContactBook()
contacts.add_contact("小明", "13800001111", "编程")
contacts.add_contact("小红", "13800002222", "画画")
contacts.add_contact("小刚", "13800003333", "编程")
contacts.list_all()
contacts.find_contact("小红")
contacts.find_by_hobby("编程")
```

### 实验3: 🥞 栈与队列大战

**零件清单**:
- 电脑
- 一摞真实盘子或书本（可选，用于直观演示）

**玩法描述**:
1. 用 Python 实现栈（后进先出）：
```python
class Stack:
    def __init__(self):
        self.items = []
    
    def push(self, item):
        self.items.append(item)
        print(f"压入：{item}，栈内：{self.items}")
    
    def pop(self):
        if not self.is_empty():
            item = self.items.pop()
            print(f"弹出：{item}，栈内：{self.items}")
            return item
        print("栈是空的！")
        return None
    
    def peek(self):
        if not self.is_empty():
            return self.items[-1]
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)

# 模拟浏览器的"后退"功能
history = Stack()
history.push("百度首页")
history.push("搜索Python")
history.push("Python教程")
history.push("第一章")
print("正在浏览第一章，点击后退...")
history.pop()
print("正在看Python教程，点击后退...")
history.pop()
```

2. 用 Python 实现队列（先进先出）：
```python
from collections import deque

class Queue:
    def __init__(self):
        self.items = deque()
    
    def enqueue(self, item):
        self.items.append(item)
        print(f"入队：{item}，队列：{list(self.items)}")
    
    def dequeue(self):
        if not self.is_empty():
            item = self.items.popleft()
            print(f"出队：{item}，队列：{list(self.items)}")
            return item
        print("队列是空的！")
        return None
    
    def is_empty(self):
        return len(self.items) == 0
    
    def size(self):
        return len(self.items)

# 模拟打印任务排队
printer = Queue()
printer.enqueue("小明的作业")
printer.enqueue("小红的手抄报")
printer.enqueue("小刚的作文")
print("开始打印...")
printer.dequeue()
printer.dequeue()
```
