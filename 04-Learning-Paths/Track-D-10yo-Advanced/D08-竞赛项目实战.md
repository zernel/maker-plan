## 🧠 核心原理简述（给家长）

**竞赛项目实战** 是将所学技能综合应用的关键环节。一个优秀的竞赛作品需要：

- **需求分析**：发现真实问题，提出创新解决方案
- **方案设计**：技术选型、架构设计、时间规划
- **原型开发**：快速实现最小可行产品(MVP)
- **迭代优化**：测试、反馈、改进的循环
- **文档展示**：工程笔记、答辩PPT、现场演示

竞赛评审维度：
- **创新性 (30%)**：想法是否新颖，是否有独特视角
- **技术性 (30%)**：技术实现的难度和完成度
- **实用性 (20%)**：是否解决真实问题
- **展示性 (20%)**：表达是否清晰，演示是否流畅

### 项目1: 🚗 智能循迹避障小车（WRC机器人大赛方向）

**零件清单**:
- ESP32 开发板
- 小车底盘（铝合金）
- 电机驱动板 L298N
- 直流减速电机 x4
- 红外循迹传感器 x5
- 超声波传感器 HC-SR04 x2
- 舵机 SG90 x1（用于超声波扫描）
- 锂电池组 7.4V
- 杜邦线、面包板

**项目描述**:
设计一辆能自主循迹、避障、并完成指定任务的智能小车。

**实现步骤**:
1. **需求分析** (1次活动)
   - 分析WRC竞赛规则
   - 确定功能需求（循迹、避障、任务执行）
   - 制定技术方案

2. **硬件搭建** (2次活动)
   - 组装底盘和电机
   - 安装传感器
   - 电路连接和测试

3. **软件开发** (3次活动)
   - 循迹算法（PID控制）
   - 避障算法（状态机设计）
   - 任务执行逻辑

4. **测试优化** (2次活动)
   - 调参优化
   - 故障排查
   - 性能测试

**核心代码框架**:
```python
from machine import Pin, PWM, time_pulse
import time

class SmartCar:
    def __init__(self):
        # 电机初始化
        self.motor_left = PWM(Pin(25), freq=1000)
        self.motor_right = PWM(Pin(26), freq=1000)
        
        # 循迹传感器
        self.ir_sensors = [Pin(i, Pin.IN) for i in [32, 33, 34, 35, 36]]
        
        # 超声波传感器
        self.trig = Pin(13, Pin.OUT)
        self.echo = Pin(14, Pin.IN)
        
        # PID参数
        self.kp = 0.5
        self.ki = 0.01
        self.kd = 0.1
        self.last_error = 0
        self.integral = 0
        
        # 状态机
        self.state = "IDLE"  # IDLE, TRACKING, OBSTACLE, TASK
    
    def get_distance(self):
        """测量距离"""
        self.trig.value(1)
        time.sleep_us(10)
        self.trig.value(0)
        
        pulse_time = time_pulse(self.echo, 1, 30000)
        distance = pulse_time * 0.034 / 2
        return distance
    
    def get_line_position(self):
        """获取循迹位置"""
        values = [sensor.value() for sensor in self.ir_sensors]
        # 计算加权平均位置
        weights = [-2, -1, 0, 1, 2]
        position = sum(v * w for v, w in zip(values, weights))
        return position
    
    def pid_control(self, error):
        """PID控制"""
        self.integral += error
        derivative = error - self.last_error
        output = self.kp * error + self.ki * self.integral + self.kd * derivative
        self.last_error = error
        return output
    
    def track_line(self):
        """循迹控制"""
        error = self.get_line_position()
        correction = self.pid_control(error)
        
        base_speed = 500
        left_speed = base_speed + correction
        right_speed = base_speed - correction
        
        self.motor_left.duty(int(left_speed))
        self.motor_right.duty(int(right_speed))
    
    def avoid_obstacle(self):
        """避障控制"""
        distance = self.get_distance()
        
        if distance < 20:  # 障碍物太近
            # 停车
            self.motor_left.duty(0)
            self.motor_right.duty(0)
            time.sleep(0.5)
            
            # 后退
            self.motor_left.duty(-500)
            self.motor_right.duty(-500)
            time.sleep(0.5)
            
            # 转向
            self.motor_left.duty(500)
            self.motor_right.duty(-500)
            time.sleep(0.8)
    
    def update(self):
        """状态机更新"""
        if self.state == "IDLE":
            pass
        elif self.state == "TRACKING":
            self.track_line()
            if self.get_distance() < 30:
                self.state = "OBSTACLE"
        elif self.state == "OBSTACLE":
            self.avoid_obstacle()
            self.state = "TRACKING"

# 主循环
car = SmartCar()
car.state = "TRACKING"

while True:
    car.update()
    time.sleep(10)
```

```python
from machine import Pin, ADC, I2C
import time
import ssd1306

class PlantCareSystem:
    def __init__(self):
        # 传感器
        self.soil_sensor = ADC(Pin(34))
        self.light_sensor = ADC(Pin(35))
        self.dht = DHT11(Pin(4))
        
        # 执行器
        self.pump = Pin(5, Pin.OUT)
        self.pump.value(0)
        
        # 显示屏
        i2c = I2C(scl=Pin(22), sda=Pin(21))
        self.display = ssd1306.SSD1306_I2C(128, 64, i2c)
        
        # 阈值设置
        self.soil_threshold = 30  # 土壤湿度阈值(%)
        self.watering_duration = 3  # 浇水时长(秒)
        
        # 状态
        self.is_watering = False
        self.last_watering = 0
    
    def read_sensors(self):
        """读取传感器数据"""
        soil_raw = self.soil_sensor.read()
        soil_percent = 100 - (soil_raw / 4095 * 100)
        
        light_raw = self.light_sensor.read()
        light_percent = light_raw / 4095 * 100
        
        self.dht.measure()
        temp = self.dht.temperature()
        humidity = self.dht.humidity()
        
        return {
            'soil': soil_percent,
            'light': light_percent,
            'temp': temp,
            'humidity': humidity
        }
    
    def should_water(self, soil_percent):
        """判断是否需要浇水"""
        return soil_percent < self.soil_threshold
    
    def water(self):
        """执行浇水"""
        if not self.is_watering:
            self.is_watering = True
            self.pump.value(1)
            time.sleep(self.watering_duration)
            self.pump.value(0)
            self.is_watering = False
            self.last_watering = time.time()
    
    def update_display(self, data):
        """更新显示"""
        self.display.fill(0)
        self.display.text(f"Soil: {data['soil']:.1f}%", 0, 0)
        self.display.text(f"Light: {data['light']:.1f}%", 0, 16)
        self.display.text(f"Temp: {data['temp']}C", 0, 32)
        self.display.text(f"Humidity: {data['humidity']}%", 0, 48)
        self.display.show()
    
    def run(self):
        """主循环"""
        while True:
            data = self.read_sensors()
            self.update_display(data)
            
            if self.should_water(data['soil']):
                self.water()
            
            time.sleep(60)  # 每分钟检测一次

# 运行系统
system = PlantCareSystem()
system.run()
```

```python
import tensorflow as tf
import numpy as np
import cv2

class GarbageClassifier:
    def __init__(self, model_path):
        self.model = tf.lite.Interpreter(model_path=model_path)
        self.model.allocate_tensors()
        
        self.input_details = self.model.get_input_details()
        self.output_details = self.model.get_output_details()
        
        self.categories = ['可回收', '有害', '厨余', '其他']
    
    def preprocess_image(self, image):
        """图像预处理"""
        image = cv2.resize(image, (224, 224))
        image = image / 255.0
        image = np.expand_dims(image, axis=0).astype(np.float32)
        return image
    
    def predict(self, image):
        """预测垃圾类型"""
        processed = self.preprocess_image(image)
        
        self.model.set_tensor(self.input_details[0]['index'], processed)
        self.model.invoke()
        
        output = self.model.get_tensor(self.output_details[0]['index'])
        predicted_class = np.argmax(output[0])
        confidence = output[0][predicted_class]
        
        return self.categories[predicted_class], confidence

# 使用
classifier = GarbageClassifier('garbage_model.tflite')
cap = cv2.VideoCapture(0)

while True:
    ret, frame = cap.read()
    if ret:
        category, confidence = classifier.predict(frame)
        print(f"识别结果：{category} ({confidence:.2%})")
        
        cv2.putText(frame, f"{category}: {confidence:.2%}", 
                   (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 1, (0, 255, 0), 2)
        cv2.imshow('Garbage Classifier', frame)
        
        if cv2.waitKey(1) & 0xFF == ord('q'):
            break

cap.release()
cv2.destroyAllWindows()
```
